rally_lengths <- rally_data |>
  filter(!is.na(rally_winner)) |>
  group_by(match_id, set, rally) |>
  slice_tail(n = 1) |>
  summarise(
    rally_length = max(stroke),
    rally_winner = rally_winner,
    .groups = "drop"
  )|>
  mutate(
    chou_result = if_else(
      rally_winner == "CHOU Tien Chen", 
      "Win", 
      "Loss"
    )
  )
